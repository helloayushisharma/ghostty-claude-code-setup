" Modern vim config - Warp/IDE-like experience

" ===== General Settings =====
set nocompatible
set encoding=utf-8
set number
set relativenumber
set cursorline
set showmatch
set mouse=a
set clipboard=unnamed
set backspace=indent,eol,start
set scrolloff=8
set sidescrolloff=8
set hidden
set nowrap
set signcolumn=yes

" No annoying sounds
set noerrorbells
set novisualbell

" Search
set ignorecase
set smartcase
set incsearch
set hlsearch

" Indentation
set tabstop=4
set shiftwidth=4
set expandtab
set smartindent
set autoindent

" Cursor (block in normal, line in insert)
let &t_SI = "\e[6 q"
let &t_EI = "\e[2 q"

" ===== macOS Keybindings (Warp/IDE-like) =====

" Option+Left/Right - Word navigation
nmap <Esc>b b
nmap <Esc>f w
imap <Esc>b <C-o>b
imap <Esc>f <C-o>w
vmap <Esc>b b
vmap <Esc>f w

" Cmd+Left/Right - Line start/end (Ctrl+A/E from terminal)
nmap <C-a> 0
nmap <C-e> $
imap <C-a> <C-o>0
imap <C-e> <C-o>$
vmap <C-a> 0
vmap <C-e> $

" Option+Backspace - Delete word backward
imap <Esc><BS> <C-w>

" Cmd+Backspace - Delete to line start (Ctrl+U from terminal)
imap <C-u> <C-o>d0

" Cmd+Z - Undo
nmap <C-z> u
imap <C-z> <C-o>u

" Cmd+Shift+Z - Redo
nmap <C-y> <C-r>
imap <C-y> <C-o><C-r>

" Cmd+S - Save (Ctrl+S)
nmap <C-s> :w<CR>
imap <C-s> <C-o>:w<CR>

" Cmd+D - Duplicate line
nmap <C-d> yyp
imap <C-d> <C-o>yy<C-o>p

" Up/Down in insert mode
imap <C-p> <Up>
imap <C-n> <Down>

" Home/End
nmap <Home> 0
nmap <End> $
imap <Home> <C-o>0
imap <End> <C-o>$

" ===== Quality of Life =====

" Return to last edit position when opening files
autocmd BufReadPost * if line("'\"") > 1 && line("'\"") <= line("$") | exe "normal! g'\"" | endif

" Auto-remove trailing whitespace on save
autocmd BufWritePre * :%s/\s\+$//e

" Faster escape from insert mode
set ttimeoutlen=10

" Better splits
set splitbelow
set splitright

" Navigate splits with Ctrl+hjkl
nmap <C-h> <C-w>h
nmap <C-j> <C-w>j
nmap <C-k> <C-w>k
nmap <C-l> <C-w>l

" Clear search highlight with Escape
nnoremap <Esc> :noh<CR><Esc>

" Tab navigation
nmap <C-t> :tabnew<CR>
nmap <Tab> :tabnext<CR>
nmap <S-Tab> :tabprev<CR>
