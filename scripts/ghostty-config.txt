# ===== Ghostty Configuration =====
# Modern terminal setup with Warp-like features

# Theme (Catppuccin Mocha - great colors for syntax highlighting)
theme = dark:Catppuccin Mocha,light:Catppuccin Latte

# macOS: Treat Option key as Alt (required for Option+key bindings)
macos-option-as-alt = true

# Appearance
background-opacity = 0.85
background-blur = true
window-padding-x = 10
window-padding-y = 8

# Bold text uses bright colors (better visibility)
bold-is-bright = true

# Session persistence (restore tabs and windows on restart)
window-save-state = always
quit-after-last-window-closed = false

# Shell Integration
shell-integration = detect
shell-integration-features = no-cursor,sudo,title

# Line Navigation (cmd+left/right for beginning/end of line)
keybind = cmd+right=text:\x05
keybind = cmd+left=text:\x01

# Word Navigation (option+left/right to skip words)
keybind = alt+right=text:\x1bf
keybind = alt+left=text:\x1bb

# Delete word backward (option+backspace)
keybind = alt+backspace=text:\x17

# Prompt Navigation (jump between prompts)
keybind = cmd+down=jump_to_prompt:1
keybind = cmd+up=jump_to_prompt:-1

# Tab Management
keybind = cmd+t=new_tab
keybind = cmd+w=close_surface
keybind = cmd+shift+left_bracket=previous_tab
keybind = cmd+shift+right_bracket=next_tab
keybind = cmd+1=goto_tab:1
keybind = cmd+2=goto_tab:2
keybind = cmd+3=goto_tab:3
keybind = cmd+4=goto_tab:4
keybind = cmd+5=goto_tab:5

# Split Panes
keybind = cmd+d=new_split:right
keybind = cmd+shift+d=new_split:down
keybind = cmd+alt+up=goto_split:top
keybind = cmd+alt+down=goto_split:bottom
keybind = cmd+alt+left=goto_split:left
keybind = cmd+alt+right=goto_split:right

# Scrolling
keybind = cmd+shift+up=scroll_page_up
keybind = cmd+shift+down=scroll_page_down
keybind = cmd+home=scroll_to_top
keybind = cmd+end=scroll_to_bottom

# Quick Actions
keybind = cmd+k=clear_screen
keybind = cmd+shift+comma=reload_config
keybind = cmd+comma=open_config
keybind = cmd+plus=increase_font_size:1
keybind = cmd+minus=decrease_font_size:1
keybind = cmd+zero=reset_font_size

# Clipboard
keybind = cmd+c=copy_to_clipboard
keybind = cmd+v=paste_from_clipboard
clipboard-paste-protection = true

# Cursor (block cursor, no blinking)
cursor-style = block
cursor-style-blink = false

# Mouse
mouse-hide-while-typing = true
copy-on-select = clipboard
